---
title: "Get Card Details"
api: "GET https://api.uncash.io/api/v1/cards/{id}"
description: "Get detailed information about a specific card including sensitive data"
---

## Path Parameters

<ParamField path="id" type="integer" required>
  The card ID to retrieve details for
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Card details">
    <ResponseField name="id" type="integer">Database card ID</ResponseField>
    <ResponseField name="created_at" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="updated_at" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="PAN" type="string">Full card number (sensitive)</ResponseField>
    <ResponseField name="EXP" type="string">Expiration date (MM/YY)</ResponseField>
    <ResponseField name="CVV" type="string">Card verification value (sensitive)</ResponseField>
    <ResponseField name="masked_PAN" type="string">Masked card number</ResponseField>
    <ResponseField name="name" type="string">Card nickname</ResponseField>
    <ResponseField name="balance" type="number">Current card balance</ResponseField>
    <ResponseField name="deposit" type="number">Total deposits made</ResponseField>
    <ResponseField name="withdraw" type="number">Total withdrawals</ResponseField>
    <ResponseField name="spent" type="number">Total amount spent</ResponseField>
    <ResponseField name="status" type="string">Card status</ResponseField>
    <ResponseField name="currency" type="string">Card currency</ResponseField>
    <ResponseField name="currency_short_name" type="string">Currency abbreviation</ResponseField>
    <ResponseField name="category" type="string">Card category</ResponseField>
    <ResponseField name="bin_id" type="string">Associated BIN</ResponseField>
    <ResponseField name="3ds" type="boolean">3D Secure enabled</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET https://api.uncash.io/api/v1/cards/123 \
  -H "Authorization: Bearer uc_your_api_token_here"
```

```javascript JavaScript
const response = await fetch('https://api.uncash.io/api/v1/cards/123', {
  headers: {
    'Authorization': 'Bearer uc_your_api_token_here'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.uncash.io/api/v1/cards/123"
headers = {"Authorization": "Bearer uc_your_api_token_here"}

response = requests.get(url, headers=headers)
data = response.json()
```

</RequestExample>
