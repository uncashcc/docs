---
title: "Get Card Details"
api: "GET https://api.uncash.io/api/v1/cards/{id}"
description: "Get detailed information about a specific card (non-sensitive data only)"
---

## Path Parameters

<ParamField path="id" type="string" required>
  The card ID (UUID) to retrieve details for
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Card details">
    <ResponseField name="id" type="string">Database card ID (UUID)</ResponseField>
    <ResponseField name="created_at" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="updated_at" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="category" type="string">Card category ("Wallet" or "Online")</ResponseField>
    <ResponseField name="masked_PAN" type="string">Masked card number (e.g., "**** **** **** 1249")</ResponseField>
    <ResponseField name="name" type="string">Card nickname</ResponseField>
    <ResponseField name="balance" type="number">Current card balance</ResponseField>
    <ResponseField name="deposit" type="number">Total deposits made to card</ResponseField>
    <ResponseField name="status" type="string">Card status ("active", "frozen", "closed")</ResponseField>
    <ResponseField name="currency" type="string">Card currency (e.g., "USD")</ResponseField>
    <ResponseField name="currency_short_name" type="string">Currency abbreviation</ResponseField>
    <ResponseField name="bin_id" type="string">Associated BIN ID</ResponseField>
    <ResponseField name="3ds" type="boolean">3D Secure enabled</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET https://api.uncash.io/api/v1/cards/card-id \
  -H "Authorization: Bearer uc_your_api_token_here"
```

```javascript JavaScript
const response = await fetch('https://api.uncash.io/api/v1/cards/card-id', {
  headers: {
    'Authorization': 'Bearer uc_your_api_token_here'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.uncash.io/api/v1/cards/card-id"
headers = {"Authorization": "Bearer uc_your_api_token_here"}

response = requests.get(url, headers=headers)
data = response.json()
```

</RequestExample>
