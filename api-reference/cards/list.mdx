---
title: "List Cards"
api: "GET https://api.uncash.io/api/v1/cards"
description: "Get all user's virtual cards with summary information"
---

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="total_balance" type="number">Total balance across all cards</ResponseField>
    <ResponseField name="total_cards" type="number">Total number of cards</ResponseField>
    <ResponseField name="cards" type="array">
      Array of user cards
      <Expandable title="Card object">
        <ResponseField name="id" type="string">Card ID</ResponseField>
        <ResponseField name="name" type="string">Card name</ResponseField>
        <ResponseField name="status" type="string">Card status (active, closed, etc.)</ResponseField>
        <ResponseField name="balance" type="number">Card balance</ResponseField>
        <ResponseField name="category" type="string">Card category (Wallet, Online)</ResponseField>
        <ResponseField name="card_plan" type="number">Card plan ID</ResponseField>
        <ResponseField name="card_number" type="string">Masked card number</ResponseField>
        <ResponseField name="created_at" type="string">ISO 8601 timestamp</ResponseField>
        <ResponseField name="updated_at" type="string">ISO 8601 timestamp</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.uncash.io/api/v1/cards" \
  -H "Authorization: Bearer uc_your_api_token_here"
```

```javascript JavaScript
const response = await fetch('https://api.uncash.io/api/v1/cards', {
  headers: {
    'Authorization': 'Bearer uc_your_api_token_here'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.uncash.io/api/v1/cards"
headers = {"Authorization": "Bearer uc_your_api_token_here"}

response = requests.get(url, headers=headers)
data = response.json()
```
