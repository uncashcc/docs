---
title: "Create Card"
api: "POST https://api.uncash.io/api/v1/cards"
description: "Create a new virtual card with automatic wallet deduction"
---

## Body Parameters

<ParamField body="bin_id" type="string" required>
  BIN identifier for the card (get from `/api/v1/bins`)
</ParamField>

<ParamField body="name" type="string" required>
  Card nickname (1-255 characters)
</ParamField>

<ParamField body="planId" type="integer" required>
  Card plan ID (get from `/api/v1/card-plans`)
</ParamField>

<ParamField body="coupon_id" type="string">
  Optional coupon ID for discounts
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the card was created successfully
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Card details">
    <ResponseField name="id" type="string">Database card ID (UUID)</ResponseField>
    <ResponseField name="category" type="string">Card category ("Wallet" or "Online")</ResponseField>
    <ResponseField name="status" type="string">Card status ("active")</ResponseField>
    <ResponseField name="balance" type="number">Initial card balance</ResponseField>
    <ResponseField name="address" type="object">
      <Expandable title="Billing address">
        <ResponseField name="city" type="string">City name</ResponseField>
        <ResponseField name="line1" type="string">Address line 1</ResponseField>
        <ResponseField name="line2" type="string | null">Address line 2 (optional)</ResponseField>
        <ResponseField name="state" type="string">State/Province</ResponseField>
        <ResponseField name="country" type="string">Country code</ResponseField>
        <ResponseField name="postalCode" type="string">Postal/ZIP code</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp
</ResponseField>

<RequestExample>

```bash cURL
curl -X POST https://api.uncash.io/api/v1/cards \
  -H "Authorization: Bearer uc_your_api_token_here" \
  -H "Content-Type: application/json" \
  -d '{
    "bin_id": "bin-id",
    "name": "My Business Card",
    "planId": 1
  }'
```

```javascript JavaScript
const response = await fetch('https://api.uncash.io/api/v1/cards', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer uc_your_api_token_here',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    bin_id: "bin-id",
    name: "My Business Card",
    planId: 1
  })
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.uncash.io/api/v1/cards"
headers = {
    "Authorization": "Bearer uc_your_api_token_here",
    "Content-Type": "application/json"
}
data = {
    "bin_id": "bin-id",
    "name": "My Business Card",
    "planId": 1
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
```
